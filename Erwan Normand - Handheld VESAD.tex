\newcommand{\twocols}[2]{% Two side-by-side colums
  \begin{columns}%
    \begin{column}{0.5\textwidth}#1\end{column}%
    \begin{column}{0.5\textwidth}#2\end{column}%
  \end{columns}%
}
\newcommand{\twocolsframe}[3]{%
  \begin{frame}{#1}%
    \twocols{#2}{#3}%
  \end{frame}%
}
\newcommand{\figuregraphic}[2][1]{%
  \includegraphics[width=#1\textwidth, height=6cm, keepaspectratio]{figures/#2}%
}
\newcommand<>{\figurecaption}[3][1]{%
  \only#4{%
    \vspace{-.5cm}
    \begin{figure}%
      \figuregraphic[#1]{#2}
      \caption{#3}%
    \end{figure}%
  }%
}
\newcommand<>{\twocolsfigure}[5]{%
  \only#6{%
    \begin{figure}%
      \twocols{%
        \centering%
        \figuregraphic[#1]{#2}%
      }{%
        \centering%
        \figuregraphic[#3]{#4}%
      }%
      \caption{#5}%
    \end{figure}%
  }%
}
\newcommand<>{\movie}[4][1]{%
  \only#5{%
    \vspace{-.5cm}%
    \begin{center}%
      \includemedia[%
        width=#1\textwidth, totalheight=6.5cm, keepaspectratio,%
        addresource=figures/#3, activate=pageopen,%
        flashvars={%
          source=figures/#3%
          &autoPlay=true%
          &loop=true%
        }]%
        {\figuregraphic[#1]{#2}}%
        {VPlayer.swf}\\%
        #4%
    \end{center}%
  }%
}
